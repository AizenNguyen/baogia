#include<iostream>
#include<ctime>
using namespace std;
int n,x,mx,spf[3000006],cnt[3000006];
long long ans=1;
int main(){
    ios::sync_with_stdio(0);cin.tie(0);
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>x;
        if(x>mx) mx=x;
        while(x>1){
            if(spf[x]==0)
                for(int i=x;i<=mx;i+=x)
                    if(spf[i]==0) spf[i]=x;
            cnt[spf[x]]++;
            x/=spf[x];
        }
    }
    for(int i=2;i<=mx;i++)
        if(cnt[i])
            ans=ans*(cnt[i]+1)%123456789;
    cout<<ans;
}
